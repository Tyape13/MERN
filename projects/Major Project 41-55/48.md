### 7 - Cookie Parser
const cookieParser = require('cookie-parser');
app.use(cookieParser());

app.get('/anypath', (req, res)=>{
    let {name= "default"} = req.cookies; // sending cookies in normal path itself is possible.
    res.send(`hello ${name}`);
});

### 8 - Signed Cookies  
we can manually change any values of any cookies, so we use signed cookies. it's like sealed water bottle or package. 

app.use(cookieParser("secretCode")); // any high complex secretCode

app.get("/getsignedcookie", (req, res)=>{
    res.cookie("greet", "namaste", {signed: true});
    res.send("signed cookie set");
});

// to verify
app.get("/verifycookie", (req, res)=>{
    console.log(req.signedCookies);
    res.send(req.signedCookies); // i think this will only give us signed cookies that's not tampered or modified or manipulated. 
})

// req.cookies will give us unsigned cookies.
// while req.signedCookies will give us signed cookies.
